    static public class StringExtension
    {
        static public DateTime ToDate(this String str)
        {
            DateTime result;
            string [] segements = str.Split('/');
            result = new DateTime(int.Parse(segements[2]), int.Parse(segements[1]), int.Parse(segements[0]));
            return result;
        }
    }
